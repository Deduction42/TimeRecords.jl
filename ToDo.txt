@recipe function f(ts::TimeSeries; use_dates=true)
    if !use_dates
        return (timestamps(ts), values(ts))
    else
        xrot --> 30

        dt = diff(TimeInterval(ts))
        if dt < 24*3600
            return (Time.(datetime.(ts)), values(ts))
        else
            return (datetimes(ts), values(ts))
        end
    end
end
